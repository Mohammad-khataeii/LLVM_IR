% Programma base + ciclo for/if

N = 3;

A = [1 2 3;
     4 5 6;
     7 8 9];

B = [1 0 0;
     0 1 0;
     0 0 1];

B(1,2) = 2;
B(2,3) = 3;

C = A + B;
D = A * B;
E = C * D;

for i = 1:3
    for j = 1:3
        if E(i,j) > 20
            E(i,j) = E(i,j) - 10;
        else
            E(i,j) = E(i,j) + 2;
        end
    end
end

disp(D);
disp(E);
