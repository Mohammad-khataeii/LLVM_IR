% Programma completo alternativo 1

N = 2;

A = [2 1;
     4 3];

B = [1 0;
     0 1];

B(1,2) = 3;
B(2,1) = 2;

C = A + B;
D = A * B;
E = C * D;

for i = 1:2
    for j = 1:2
        if E(i,j) > 25
            E(i,j) = E(i,j) - 5;
        else
            E(i,j) = E(i,j) + 1;
        end
    end
end

disp(D);
disp(E);
